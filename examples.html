Beispiel 1: Scrollen nach rechts

Hier wird mit Hilfe des Parallax-Effekts ein Fisch beim Scrollen nach rechts schwimmen.

<div class="parallax">
<img src="images/Fisch1.png" id="fisch1" />
</div>

.parallax {
position: relative;
display: flex;
overflow: hidden;
justify-content: center;
align-self: center;
height: 100vh;
background-size: cover;
background-position: center;
background-attachment: fixed;
}
 
.parallax img {
position: absolute;
top: 0px;
left: 0px;
pointer-events: none;
}

let fisch1 = document.getElementById("fisch1");
 
window.addEventListener("scroll", () => {
let scrollHeight = window.scrollY;
let windowWidth = window.innerWidth;
let windowHeight = window.innerHeight;
let parallaxStay = scrollHeight + "px";
let parallaxMoveRight = (scrollHeight / windowHeight) * windowWidth + "px";
 
fisch1.style.transform = "translateX(" + parallaxMoveRight + ")";
fisch1.style.transform = "translateY(" + parallaxStay + ")";
});

display: flex;

align-self: center;

background-size: cover;
background-position: center;

.parallax img {
position: absolute;
top: 0px;

pointer-events: none;
}

let fisch1 = document.getElementById("fisch1");
let bubble1 = document.getElementById("blasen1");
let bubble2 = document.getElementById("blasen2");

 
window.addEventListener("scroll", () => {
let scrollHeight = window.scrollY;
let parallaxMoveRight = scrollHeight + (1 / 10) * scrollHeight +"px";

bubble1.style.transform = "translateY(" + parallaxMoveUp + ")";
